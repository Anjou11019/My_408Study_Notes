例子2：使用variadic template重写printf()

```cpp
void printf(const char *s){
    while(*s){
        if(*s=='%'&&*(++s) != '%')
            throw std::runtime_error("invalid format string : missing arguments");
        std::cout << *s++;
    }
}
template<typename T,typename... Args>
void printf(const char *s,T value,Args... args){
    while(*s){
```

##### 写一个 move aware class

```cpp
#ifndef MY_STRING_H
#define MY_STRING_H

#include <cstddef>
#include <cstring>
#include <string>

class MyString
{
public:
    static size_t DCtor; // 累计默认构造函数调用次数
    static size_t Ctor;  // 累计构造函数调用次数
    static size_t CCtor; // 累计拷贝构造调用次数
    static size_t CAsgn; // 累计copy assignment调用次数
    static size_t MCtor; // 累计move构造函数调用次数
    static size_t MAsgn; // 累计move assignment调用次数
    static size_t Dtor;  // 累计析构函数调用次数
private:
    char *_data;
    size_t _len;
    void _init_data(const char *s)
    {
        _data = new char[_len + 1];
        memcpy(_data, s, _len);
        _data[_len] = '\0';
    }

public:
    // default constructor
    MyString() : _data(nullptr), _len(0) { ++DCtor; }
    // constructor
    MyString(const char *p) : _len(strlen(p))
    {
        ++Ctor;
        _init_data(p);
    }
    // copy constructor
    MyString(const MyString &str) : _len(str._len)
    {
        ++CCtor;
        _init_data(str._data);
    }
    // copy assignment
    MyString &operator=(const MyString &str)
    {
        ++CAsgn;
        if (this != &str)
        {
            if (_data)
                delete _data;
            _len = str._len;
            _init_data(str._data);
        }
        else
        {
        }
        return *this;
    }

    // move constructor，with "noexcept"
    MyString(MyString &&str) noexcept
        : _data(str._data), _len(str._len) // 把指针和len都“偷”过来
    {
        ++MCtor;
        str._data = nullptr; // 重要
        // 如果不写，则两个指针指向data，临时对象生命结束的时候调用析构函数，会导致数据被删除
        str._len = 0;
    }
    // move assignment，with "noexcept"
    MyString &operator=(MyString &&str) noexcept
    {
        ++MAsgn;
        if (this != &str)
        { // 自赋值检查
            if (_data)
                delete _data;
            _len = str._len;
            _data = str._data;
            str._len = 0;
            str._data = nullptr; // 重要
        }
        else
        {
        }
        return *this;
    }
    // destructor
    virtual ~MyString()
    {
        ++Dtor;
        if (_data)
        {
            delete _data;
        }
    }
    bool operator<(const MyString &rhs) const//为了set
    {
        return std::string(this->_data) < std::string(rhs._data);
    }
    bool operator==(const MyString &rhs) const{
        return std::string(this->_data) == std::string(rhs._data);
    }
    char* get() const{return _data;}
};
size_t MyString::DCtor = 0;
size_t MyString::Ctor = 0;
size_t MyString::CCtor = 0;
size_t MyString::CAsgn = 0;
size_t MyString::MCtor = 0;
size_t MyString::MAsgn = 0;
size_t MyString::Dtor = 0;

namespace std{
template<>
    struct hash<MyString>{
        size_t operator()(const MyString& s){
            return std::hash<std::string>()(s.get());
        }
    };
};

#endif /* MY_STRING_H */
```

##### RAII

RAII **资源获取即初始化（Resourse Acquisition Is Initialization）**

是一种C++特有的资源管理方式，依托栈和析构函数来对所有资源（包含堆内存），进行管理。提供了一种异常安全的资源管理方式，对于RAII的使用，是C++不必需要类似Java那样的垃圾收集方法，也能有效的对内存进行管理

**它将在使用前获取的资源的生命周期和某个对象的生命周期进行绑定**，确保在控制对象的生命周期结束时，按照资源获取相反顺序释放所有资源

同样的，若资源获取失败（构造函数退出并附带异常），则按照初始化的相反顺序释放所有已完全构造的成员和基类子对象所获取的资源

其**核心思想就是：利用栈上局部变量的自动析构保证资源一定会被释放**

##### RAII类的实现步骤

1.设计一个类封装资源

2.在构造函数中执行资源的初始化

3.在析构函数中执行销毁操作

4.使用时声明一个该对象的类，一般在你希望的作用域声明即可

##### C/C++的常见内存错误

野指针：未初始化或已经被释放的指针

空指针：指向空地址的指针

内存泄漏：如果使用完动态分配内存后的内存未释放，就会造成内存泄漏，长时间运行的程序会消耗大量内存

悬空指针：指向已经释放的内存的指针

内存泄漏和悬空指针的混合：有时候二者会同时存在，程序出现异常行为

# 3.16

##### shared_ptr

共享所有权，它允许**多个shared_ptr指向同一个对象**，当**最后一个**shared_ptr超出作用域时，所指向的内存才会被自动释放

每个shared_ptr的引用计数都会记录指向该对象的shared_ptr的数量，通过use_count方法可以获取shared_ptr对象的共享计数值

其创造了一个**计数器**与类对象所指的内存相关联

copy则计数器加一，销毁则计数器减一，api为use_count()

```cpp
    //常量类型
    shared_ptr<int> i_p1 = make_shared<int>(10);
    cout << "value" << *i_p1 << endl;
    cout << "used count： " << i_p1.use_count() << endl;

    //copy
    shared_ptr<int> i_p2(i_p1);
    cout << "value" << *i_p2 << endl;
    cout << "used count： " << i_p2.use_count() << endl;

    *i_p2 = 20;
    cout << "value" << *i_p1 << endl;//改2，1也会变

    i_p2 = nullptr;
    cout << "value" << *i_p1 << endl;//只清理了2，1还在
    //cout << "value" << *i_p2 << endl;

    //自定义类型
    shared_ptr<Cat> c_p1 = make_shared<Cat>();
    cout << "c_p1.use_count" << c_p1.use_count() << endl;
    shared_ptr<Cat> c_p2 = c_p1;
    shared_ptr<Cat> c_p3(c_p2);

    cout << "c_p1.use_count" << c_p1.use_count() << endl;
    cout << "c_p2.use_count" << c_p2.use_count() << endl;
    cout << "c_p3.use_count" << c_p3.use_count() << endl;

    c_p1.reset();
    cout << "c_p1.use_count" << c_p1.use_count() << endl;//0
    cout << "c_p2.use_count" << c_p2.use_count() << endl;//2
    cout << "c_p3.use_count" << c_p3.use_count() << endl;//2
```

##### shared_ptr 与 unique_ptr

不能将 <mark>shared_ptr</mark> 转化为<mark> unique_ptr</mark>

<mark>unique_ptr</mark> 可以转化为 <mark>shared_ptr</mark>： 通过std::move

**常见的设计**：

将你的函数返回为unique_ptr是一种常见的设计模式，这样可以提高代码的复用度，你可以随时改变为shared_ptr

```cpp
unique_ptr<Cat> get_unique_ptr() {
    unique_ptr<Cat> cat_p = make_unique<Cat>("local cat");
    return cat_p;
}
//shared_ptr 与 unique_ptr
unique_ptr<Cat> c_p1 = make_unique<Cat>("dd");
shared_ptr<Cat> c_p2 = move(c_p1);
cout << "c_p2 used;" << c_p2.use_count() << endl;

shared_ptr c_p3 = get_unique_ptr();//可以接收返回值为unique的指针
if (c_p3) {
    c_p3->cat_info();
    cout << "c_p3 used;" << c_p3.use_count() << endl;
}
```

##### weak_ptr

weak_ptr并<mark>不拥有所有权</mark>

并不能调用->和\*解引用

**为什么会存在：**

    A类中有一个需求需要存储其他A类对象的信息

    如果使用shared_ptr，那么销毁时会遇到循环依赖的问题（cycle dependence problem）

```cpp
    //循环依赖问题
    shared_ptr<Cat> c3 = make_shared<Cat>("c3");
    shared_ptr<Cat> c4 = make_shared<Cat>("c4");
    c3->set_friend(c4);
    c4->set_friend(c3);//程序结束后c3,c4的销毁函数未调用
```

            非常非常严重的bug，并且难以发现（因为编译会正常通过）

只需修改：

```cpp
#ifndef _CAT_H_
#define _CAT_H_

...
    void set_friend(std::shared_ptr<Cat> c) {
        m_friend = c;
    }
private:
    std::string name{ "Mimi" };
    std::weak_ptr<Cat> m_friend;//改为weak即可
};

#endif
```

    weak_ptr可以使用lock()函数提升为shared_ptr（类型转换）

```cpp
    shared_ptr<Cat> s_p_c1 = make_shared<Cat>("c1");
    weak_ptr<Cat> w_p_c1(s_p_c1);

    //use_count
    cout << "use_count: " << s_p_c1.use_count() << endl;//1
    cout << "use_count: " << w_p_c1.use_count() << endl;//1 weak_ptr可以调用use_count，但计数器不加1
    //lock
    shared_ptr<Cat> s_p_c2 = w_p_c1.lock();
    cout << "use_count: " << s_p_c1.use_count() << endl;//2
    cout << "use_count: " << w_p_c1.use_count() << endl;//2
    cout << "use_count: " << s_p_c2.use_count() << endl;//2
```
